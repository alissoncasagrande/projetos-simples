#anotacoes sobre o git/github

#configuracoes do git na maquina local
git config --global user.name "Alisson Casagrande"
git config --global user.email "seu-email-do-github@example.com"


git init
git branch -M main  -> caso crie com o nome de branch errado (master)
git remote add origin https://github.com/alissoncasagrande/projetos-simples.git

git add .
git commit -m "Primeiro commit"
git push --set-upstream origin main OU git push -u origin main

git pull            -> trazer aquivos do repositorio para a maquina

git status          -> status dos arquivos (stage ou commit)
git log --oneline   -> historico de commits
git branch          -> nome do branch atual
git remote -v       -> mostra a URL associada ao origin
